cmake_minimum_required(VERSION 3.28.1)
project(game C)

# Add the GLAD library
add_library(glad STATIC src/glad.c)
file(GLOB_RECURSE source_files "src/*.c")

add_executable(game ${source_files})

set(LIB lib)
set(INCLUDE include)
include_directories(${INCLUDE})

set_target_properties(game PROPERTIES OUTPUT_NAME "${PROJECT_SOURCE_DIR}/bin/Game")
#link_directories(${LIB})

# Add the GLFW DLL as a library
add_library(glfw3dll SHARED IMPORTED)
set_target_properties(glfw3dll PROPERTIES IMPORTED_IMPLIB "lib/glfw3.lib")

set_target_properties(glfw3dll PROPERTIES IMPORTED_LOCATION ${CMAKE_CURRENT_SOURCE_DIR}/libs/glfw3.dll)

# Link against GLFW
target_link_libraries(game PRIVATE glfw3dll)

target_include_directories(game PRIVATE ${INCLUDE})
